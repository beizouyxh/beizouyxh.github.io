# 绝对定位+margin

            width: 200px;
            height: 200px;
            border: lime solid 1px;
            position: relative;
        }
   .children-box{
            position: absolute;
            width: 100px;
            height: 100px;
            left: 50%;
            top: 50%;
            margin: -50px 0 0 -50px;
            background-color: limegreen;
    }

            height: 200px;

            width: 100px;

            left: 50%;
            top: 50%;
            background-color: lime;
            transform: translate(-50%,-50%);
   }
